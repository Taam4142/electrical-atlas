## Route wrapper checklist

Each published lesson also needs an Astro route wrapper:

Before using a new `<lesson-key>`, add that key and its curated suggestions in:

```text
electrical-atlas-site/src/lib/suggestions.ts
```

```astro
---
import BaseLayout from "../../../layouts/BaseLayout.astro";
import LessonShell from "../../../components/LessonShell.astro";
import SuggestionSection from "../../../components/SuggestionSection.astro";
import Content from "../../../content/lessons/en/<slug>.mdx";
import { atlasTopics } from "../../../lib/generated/atlasTopics";
import { getLessonSuggestions } from "../../../lib/suggestions";

const suggestions = getLessonSuggestions("<lesson-key>", "en", atlasTopics);
---

<BaseLayout
  title="<Page title>"
  description="<Page description>"
  lang="en"
  counterpartPath="/th/lessons/<slug>/"
  counterpartLabel="Thai"
>
  <LessonShell
    locale="en"
    eyebrow="<lesson type>"
    title="<Lesson title>"
    subtitle="<One-sentence promise>"
    lessonId="ea.lesson.<slug>.v0.1"
  >
    <div class="lesson-content">
      <Content />
    </div>
    <SuggestionSection locale="en" items={suggestions} />
  </LessonShell>
</BaseLayout>
```
